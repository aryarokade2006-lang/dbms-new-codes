CREATE TABLE dept13(
deptno NUMBER(4) PRIMARY KEY,
dname VARCHAR2(30) NOT NULL,
loc VARCHAR2(30)
);

CREATE TABLE emp13(
empno NUMBER(6) PRIMARY KEY,
ename VARCHAR2(25) NOT NULL,
job VARCHAR2(10) NOT NULL,
mgr NUMBER(6),
sal NUMBER(8,2),
comm NUMBER(2,2),
deptno NUMBER(4),
FOREIGN KEY(deptno) REFERENCES dept13(deptno)
);

CREATE TABLE salgrade13(
grade NUMBER,
losal NUMBER,
hisal NUMBER
);

SELECT ename, job, dname, sal, grade
FROM emp13 e
JOIN dept13 d
ON e.deptno = d.deptno
JOIN salgrade13 s
ON e.sal BETWEEN s.losal AND s.hisal;

SELECT ename, sal, mgr, dname
FROM emp13 e
JOIN dept13 d
ON e.deptno = d.deptno
WHERE sal > 2000;

SELECT empno, ename, job, e.deptno, dname, loc
FROM emp13 e
LEFT OUTER JOIN dept13 d
ON e.deptno = d.deptno;

SELECT ename, dname
FROM emp13 e
RIGHT OUTER JOIN dept13 d
ON e.deptno = d.deptno;

SELECT empno, dname, loc
FROM emp13 e
FULL OUTER JOIN dept13 d
ON e.deptno = d.deptno;
