CREATE TABLE dept(
deptno NUMBER(10) PRIMARY KEY,
dname VARCHAR2(20),
loc VARCHAR2(20)
);

CREATE TABLE emp(
empno NUMBER(10) PRIMARY KEY,
ename VARCHAR2(20),
job VARCHAR2(20),
mgr NUMBER(10),
hiredate DATE,
sal NUMBER(10,2),
comm NUMBER(10,2),
deptno NUMBER(10),

FOREIGN KEY(deptno)
REFERENCES dept(deptno)
);

SELECT empno, ename, sal, dname, loc
FROM emp
NATURAL JOIN dept;

SELECT job, mgr, sal, comm, dname
FROM emp
JOIN dept
ON emp.deptno = dept.deptno
WHERE job = 'SALESMAN';

SELECT ename, job, emp.deptno, dname
FROM emp
JOIN dept
ON emp.deptno = dept.deptno
WHERE loc = 'DALLAS';

SELECT empno, ename, job, mgr, emp.deptno, dname
FROM emp
LEFT JOIN dept
ON emp.deptno = dept.deptno
WHERE loc = 'DALLAS'
OR mgr IS NULL
ORDER BY empno;
