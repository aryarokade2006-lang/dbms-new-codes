SET SERVEROUTPUT ON;

DECLARE

CURSOR c1 IS
SELECT *
FROM emp21
ORDER BY sal DESC;

v_empno emp21.empno%TYPE;

v_ename emp21.ename%TYPE;

v_sal emp21.sal%TYPE;

n NUMBER := &n;

i NUMBER := 0;

BEGIN

OPEN c1;

LOOP

FETCH c1
INTO v_empno,v_ename,v_sal;

EXIT WHEN c1%NOTFOUND;

i := i + 1;

IF i <= n THEN

DBMS_OUTPUT.PUT_LINE(
v_empno || ' ' ||
v_ename || ' ' ||
v_sal
);

END IF;

END LOOP;

CLOSE c1;

END;
